   258→/** Countdown timer display */
   259→function CountdownTimer({ targetDate, language }: { targetDate: string; language: string }) {
   260→  const [timeLeft, setTimeLeft] = useState("");
   261→  const isFr = language === "fr";
   262→
   263→  useEffect(() => {
   264→    const update = () => {
   265→      const now = new Date().getTime();
   266→      const target = new Date(targetDate).getTime();
   267→      const diff = target - now;
   268→
   269→      if (diff <= 0) {
   270→        setTimeLeft(isFr ? "En cours" : "Live");
   271→        return;
   272→      }
   273→
   274→      const hours = Math.floor(diff / (1000 * 60 * 60));
   275→      const minutes = Math.floor((diff % (1000 * 60 * 60)) / (1000 * 60));
   276→      const seconds = Math.floor((diff % (1000 * 60)) / 1000);
   277→
   278→      if (hours > 24) {
   279→        const days = Math.floor(hours / 24);
   280→        setTimeLeft(`${days}d ${hours % 24}h`);
   281→      } else if (hours > 0) {
   282→        setTimeLeft(`${hours}h ${minutes}m`);
   283→      } else {
   284→        setTimeLeft(`${minutes}m ${seconds}s`);
   285→      }
   286→    };
   287→
   288→    update();
   289→    const interval = setInterval(update, 1000);
   290→    return () => clearInterval(interval);
   291→  }, [targetDate, isFr]);
   292→
   293→  return <span className="text-[10px] font-mono font-bold text-primary">{timeLeft}</span>;
   294→}

   300→function CalendarSubTab({ language }: { language: string }) {
   301→  const isFr = language === "fr";
   302→  const [calendarData, setCalendarData] = useState<any>(null);
   303→  const [loading, setLoading] = useState(true);
   304→  const [error, setError] = useState<string | null>(null);
   305→  const [period, setPeriod] = useState<PeriodFilter>("today");
   306→  const intervalRef = useRef<NodeJS.Timeout | null>(null);
   307→
   308→  const fetchCalendar = useCallback(async () => {
   309→    try {
   310→      const res = await fetch(`/api/market/calendar?lang=${language}&period=${period}`);
   311→      if (!res.ok) throw new Error("Fetch failed");
   312→      const data = await res.json();
   313→      setCalendarData(data);
   314→      setError(null);
   315→    } catch {
   316→      setError(isFr ? "Erreur lors du chargement du calendrier" : "Error loading calendar");
   317→    } finally {
   318→      setLoading(false);
   319→    }
   320→  }, [language, isFr, period]);
   321→
   322→  // Reset loading when period changes
   323→  useEffect(() => {
   324→    setLoading(true);
   325→  }, [period]);
   326→
   327→  useEffect(() => {
   328→    fetchCalendar();
   329→    intervalRef.current = setInterval(fetchCalendar, 5 * 60 * 1000);
   330→    return () => {
   331→      if (intervalRef.current) clearInterval(intervalRef.current);
   332→    };
   333→  }, [fetchCalendar]);
   334→
   335→  // Normalize events from API format to component format
   336→  const normalizedEvents = useMemo(() => {
   337→    const rawEvents: any[] = calendarData?.events || [];
   338→    return rawEvents.map((evt: any) => {
   339→      // Build a date from today + time
   340→      const today = new Date();
   341→      let eventDate: Date;
   342→      if (evt.date) {
   343→        eventDate = new Date(evt.date);
   344→      } else if (evt.time) {
   345→        const [h, m] = evt.time.split(":").map(Number);
   346→        eventDate = new Date(today.getFullYear(), today.getMonth(), today.getDate(), h || 0, m || 0);
   347→      } else {
   348→        eventDate = today;
   349→      }
   350→
   351→      return {
   352→        title: evt.event || evt.title || "",
   353→        date: eventDate.toISOString(),
   354→        time: evt.time || "",
   355→        currency: (evt.currency || "").toUpperCase(),
   356→        impact: (evt.impact || "low").toUpperCase(),
   357→        countryFlag: evt.country || "🌍",
   358→        forecast: evt.forecast || null,
   359→        previous: evt.previous || null,
   360→        actual: evt.actual || null,
   361→      };
   362→    });
   363→  }, [calendarData]);
   364→
   365→  // Find next high-impact event
   366→  const nextHighImpact = useMemo(() => {
   367→    const now = new Date();
   368→    return normalizedEvents.find(e => e.impact === "HIGH" && new Date(e.date) > now) || null;
   369→  }, [normalizedEvents]);
   370→
   371→  // Week events: use normalizedEvents directly when period=week (API already returns correct period)
   372→  // For period=today fallback, filter by date range
   373→  const weekEvents = useMemo(() => {
   374→    if (period === "week") {
   375→      // API already returns the correct events for the week - just sort them
   376→      return [...normalizedEvents].sort((a: any, b: any) => new Date(a.date).getTime() - new Date(b.date).getTime());
   377→    }
   378→    // Fallback: filter by local date range (used when period=today but showing week view)
   379→    const now = new Date();
   380→    const dayOfWeek = now.getDay();
   381→    const mondayOffset = dayOfWeek === 0 ? -6 : 1 - dayOfWeek;
   382→    const monday = new Date(now.getFullYear(), now.getMonth(), now.getDate() + mondayOffset);
   383→    const friday = new Date(monday);
   384→    friday.setDate(monday.getDate() + 5);
   385→
   386→    return normalizedEvents.filter((e: any) => {
   387→      const d = new Date(e.date);
   388→      return d >= monday && d < friday;
   389→    }).sort((a: any, b: any) => new Date(a.date).getTime() - new Date(b.date).getTime());
   390→  }, [normalizedEvents, period]);
   391→
   392→  // Today's events
   393→  const todayEvents = useMemo(() => {
   394→    const now = new Date();
   395→    return normalizedEvents.filter((e: any) => {
   396→      const d = new Date(e.date);
   397→      return d.toDateString() === now.toDateString();
   398→    });
   399→  }, [normalizedEvents]);
   400→
   401→  if (loading) {
   402→    return (
   403→      <div className="space-y-4">
   404→        <div className="grid grid-cols-3 gap-3">
   405→          <Card className="p-3 sm:p-4"><Skeleton className="h-8 w-12 mx-auto mb-1" /><Skeleton className="h-3 w-16 mx-auto" /></Card>
   406→          <Card className="p-3 sm:p-4"><Skeleton className="h-8 w-12 mx-auto mb-1" /><Skeleton className="h-3 w-16 mx-auto" /></Card>
   407→          <Card className="p-3 sm:p-4"><Skeleton className="h-8 w-12 mx-auto mb-1" /><Skeleton className="h-3 w-16 mx-auto" /></Card>
   408→        </div>
   409→        <Card className="p-4 sm:p-6">
   410→          <Skeleton className="h-6 w-48 mb-4" />
   411→          <div className="space-y-3">
   412→            {Array.from({ length: 6 }).map((_, i) => (
   413→              <div key={i} className="flex items-center gap-3">
   414→                <Skeleton className="h-4 w-14" />
   415→                <Skeleton className="h-4 w-6" />
   416→                <Skeleton className="h-4 w-10" />
   417→                <Skeleton className="h-4 flex-1" />
   418→              </div>
   419→            ))}
   420→          </div>
   421→        </Card>
   422→      </div>
   423→    );
   424→  }
   425→
   426→  if (error) {
   427→    return (
   428→      <Card className="p-12 text-center">
   429→        <AlertTriangle className="w-12 h-12 mx-auto text-muted-foreground/30 mb-4" />
   430→        <p className="text-muted-foreground">{error}</p>
   431→        <Button onClick={fetchCalendar} variant="outline" className="mt-4 gap-2">
   432→          <RefreshCw className="w-4 h-4" />
   433→          {isFr ? "Réessayer" : "Retry"}
   434→        </Button>
   435→      </Card>
   436→    );
   437→  }
   438→
   439→  const highCount = normalizedEvents.filter((e: any) => e.impact === "HIGH").length;
   440→  const mediumCount = normalizedEvents.filter((e: any) => e.impact === "MEDIUM").length;
   441→
   442→  return (
   443→    <div className="space-y-4">
   444→      {/* Period Filter */}
   445→      <PeriodFilterButtons period={period} setPeriod={setPeriod} language={language} />
   446→
   447→      {/* Stats row */}
   448→      <div className="grid grid-cols-3 gap-3">
   449→        <Card className="p-3 sm:p-4 text-center">
   450→          <div className="text-2xl font-bold text-loss">{highCount}</div>
   451→          <div className="text-[10px] text-muted-foreground">{isFr ? "Impact Élevé" : "High Impact"}</div>
   452→        </Card>
   453→        <Card className="p-3 sm:p-4 text-center">
   454→          <div className="text-2xl font-bold text-amber-500">{mediumCount}</div>
   455→          <div className="text-[10px] text-muted-foreground">{isFr ? "Impact Modéré" : "Medium Impact"}</div>
   456→        </Card>
   457→        <Card className="p-3 sm:p-4 text-center">
   458→          <div className="text-2xl font-bold text-muted-foreground">{normalizedEvents.length}</div>
   459→          <div className="text-[10px] text-muted-foreground">{isFr ? "Total" : "Total"}</div>
   460→        </Card>
   461→      </div>
   462→
   463→      {/* Next high-impact event highlight */}
   464→      {nextHighImpact && (
   465→        <Card className="p-4 border-2 border-loss/30 bg-gradient-to-r from-loss/5 to-transparent">
   466→          <div className="flex items-center gap-2 mb-2">
   467→            <Zap className="w-4 h-4 text-loss" />
   468→            <span className="text-xs font-bold text-loss">
   469→              {isFr ? "Prochain événement à fort impact" : "Next High-Impact Event"}
   470→            </span>
   471→          </div>
   472→          <div className="flex items-center justify-between gap-2">
   473→            <div className="min-w-0">
   474→              <h4 className="text-sm font-bold">{nextHighImpact.title}</h4>
   475→              <div className="flex items-center gap-2 mt-1 flex-wrap">
