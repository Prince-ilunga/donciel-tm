    61→function DirectionIcon({ direction, size = "sm" }: { direction: string; size?: "sm" | "md" | "lg" }) {
    62→  const s = size === "sm" ? "w-3.5 h-3.5" : size === "md" ? "w-4 h-4" : "w-5 h-5";
    63→  const isUp = direction?.toUpperCase().includes("HAUSS") || direction?.toUpperCase().includes("BULL");
    64→  const isDown = direction?.toUpperCase().includes("BAISS") || direction?.toUpperCase().includes("BEAR");
    65→
    66→  if (isUp) return <ArrowUpRight className={cn(s, "text-profit")} />;
    67→  if (isDown) return <ArrowDownRight className={cn(s, "text-loss")} />;
    68→  return <MinusCircle className={cn(s, "text-amber-500")} />;
    69→}

    71→function DirectionBadge({ direction, confidence }: { direction: string; confidence: string }) {
    72→  const isUp = direction?.toUpperCase().includes("HAUSS") || direction?.toUpperCase().includes("BULL");
    73→  const isDown = direction?.toUpperCase().includes("BAISS") || direction?.toUpperCase().includes("BEAR");
    74→  const isNeutral = !isUp && !isDown;
    75→
    76→  const confLevel = confidence?.toLowerCase() || "moyen";
    77→  const confColor = confLevel.includes("élev") || confLevel.includes("high")
    78→    ? "text-profit" : confLevel.includes("fai") || confLevel.includes("low")
    79→    ? "text-muted-foreground" : "text-amber-500";
    80→
    81→  return (
    82→    <div className="flex items-center gap-2">
    83→      <Badge
    84→        className={cn(
    85→          "text-xs font-bold gap-1 px-3 py-1",
    86→          isUp && "bg-profit/15 text-profit border-profit/30",
    87→          isDown && "bg-loss/15 text-loss border-loss/30",
    88→          isNeutral && "bg-amber-500/15 text-amber-500 border-amber-500/30"
    89→        )}
    90→      >
    91→        {isUp && <TrendingUp className="w-3.5 h-3.5" />}
    92→        {isDown && <TrendingDown className="w-3.5 h-3.5" />}
    93→        {isNeutral && <Minus className="w-3.5 h-3.5" />}
    94→        {direction || "N/A"}
    95→      </Badge>
    96→      <span className={cn("text-[10px] font-medium", confColor)}>

    98→      </span>

   101→}

   125→function SentimentGauge({ sentiment }: { sentiment: string }) {
   126→  const s = sentiment?.toLowerCase() || "neutre";
   127→  const isVeryBull = s.includes("très haus") || s.includes("very bull") || s.includes("strongly bull");
   128→  const isBull = s.includes("haus") || s.includes("bull") || s.includes("acheteur");
   129→  const isVeryBear = s.includes("très bai") || s.includes("very bear") || s.includes("strongly bear");
   130→  const isBear = s.includes("bai") || s.includes("bear") || s.includes("vendeur");
   131→
   132→  const value = isVeryBull ? 90 : isBull ? 70 : isVeryBear ? 10 : isBear ? 30 : 50;
   133→  const label = isVeryBull ? (s.includes("haus") ? "Très Haussier" : "Very Bullish")
   134→    : isBull ? (s.includes("haus") ? "Haussier" : "Bullish")
   135→    : isVeryBear ? (s.includes("bai") ? "Très Baissier" : "Very Bearish")
   136→    : isBear ? (s.includes("bai") ? "Baissier" : "Bearish")
   137→    : (s.includes("neutre") ? "Neutre" : "Neutral");
   138→
   139→  const color = value >= 70 ? "bg-profit" : value >= 55 ? "bg-profit/60" : value <= 30 ? "bg-loss" : value <= 45 ? "bg-loss/60" : "bg-amber-500";
   140→
   141→  return (
   142→    <div className="space-y-1.5">
   143→      <div className="flex items-center justify-between">
   144→        <span className="text-[10px] font-medium text-muted-foreground">Sentiment</span>
   145→        <span className={cn("text-[10px] font-bold",
   146→          value >= 55 ? "text-profit" : value <= 45 ? "text-loss" : "text-amber-500"
   147→        )}>{label}</span>
   148→      </div>
   149→      <div className="h-2 rounded-full bg-muted overflow-hidden">
   150→        <div
   151→          className={cn("h-full rounded-full transition-all duration-700", color)}
   152→          style={{ width: `${value}%` }}
   153→        />
   154→      </div>
   155→    </div>
   156→  );
   157→}
